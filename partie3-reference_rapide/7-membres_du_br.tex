\subsubsection*{Description rapide des postes}

\begin{description}

  \item[Prez]{Grand manitou qui passe sa journée à la DSI à négocier pour entretenir les accès des élèves au réseau.}

  \item[Trez]{Escroc de service qui cherche uniquement à avoir un compte en banque le plus rempli possible pour pouvoir organiser un voyage de geeks à Redmond, euh, plutôt Cupertino\ldots en fait je sais plus.}

  \item[Com]{Assistante du prez pour les relations avec les entreprises.}

  \item[root]{Les roots sont les administrateurs du réseau. Ce sont eux qui s'évertuent à maintenir en état de marche les serveurs, à rajouter de nouveaux services et à repérer les boulets qui font de la merde sur le réseau.}

  \item[admin@windows]{Administrateurs du domaine Windows. Ce sont les responsables de l'antivirus fourni par le BR\ldots donc en cas de virus ce sont les mieux placés pour répondre.}

  \item[devel]{Joyeux programmeurs qui sont là pour améliorer les logiciels du BR --- \app{qRezix} et ses plug-ins, le site web \url{http://frankiz/}. Ce sont eux qui tous les deux mois te disent que ton qRezix n'est pas à jour.}

  \item[news]{Mainteneurs du serveur de news, ils surveillent aussi que tu ne postes pas n'importe quoi et que tu respectes bien les règles de base comme les crossposts (marteau-thérapie);}

  \item[web@frankiz]{Webmestres de frankiz, ils valident les annonces et les activités et surveillent le contenu du site de ton binet ou de ton site perso.}

  \item[X-share]{Personnes sympathique qui cherche à longueur de temps de nouveaux logiciels (libres) à proposer aux élèves dans la section Télécharger de Frankiz}

  \item[FAQ]{Ils sont là rassembler toutes les questions que tu peux imaginer pour remplir la FAQ (Foire aux Questions) de Frankiz. N'hésite pas à leur donner des idées !}

  \item[InfoBR]{Boulet qui s'est trouvé un poste pipo et qui dit à tous les autres d'écrire. Par contre, si vous trouvez des bêtises ici, c'est de sa faute, donc vous pouvez aller le taper :)}

  \item[TV]{Changeurs de chaîne, dieux du multicast, ils ont souvent envie de taper les switchs de la DSI}

  \item[QDJ Master]{Chaque jour une question sur Frankiz... n'hésitez pas à faire vos propositions à \url{qdj@frankiz}}

  \item[IRCop]{Responsable des relations avec RezoSup. Viendez sur IRC !}

  \item[TOL]{Vérificateur de photos... mais surtout se tape la surveillance du Trombi On Line}

\end{description}
