\subsubsection*{Description rapide des postes}

\begin{description}

  \item[Prez]{Grand manitou qui passe sa journée à la DSI à négocier pour entretenir les accès des élèves au réseau.}

  \item[Trez]{Escroc de service qui cherche uniquement à remplir au maximum le compte en banque pour organiser un voyage de geeks à Redmond, ou plutôt Cupertino\ldots}

  \item[Com]{Assistante du prez pour les relations avec les entreprises.}

  \item[root]{Les roots sont les administrateurs du réseau. Ce sont eux qui s'évertuent à maintenir en état de marche les serveurs, à rajouter de nouveaux services et à repérer les boulets qui font de la merde sur le réseau. Écris leur à \mail{root@frankiz}, ou à \mail{binets@frankiz} s'il s'agit de gérer un compte de binet.}

  \item[admin@windows] {(\mail{windows@frankiz}) Administrateurs du domaine Windows. En cas de problème avec Windows, en particulier avec l'antivirus, ce sont les mieux placés pour t'aider ; bien sûr c'est plus facile si tu es sur le domaine !}

  \item[devel]{Joyeux programmeurs qui sont là pour améliorer les logiciels du BR --- \app{qRezix} et ses plug-ins (\mail{qrezix@frankiz}), le site web \url{http://frankiz/}. Ce sont eux qui tous les deux mois te disent que ton qRezix n'est pas à jour.}

  \item[news] {(\mail{news@frankiz}) Mainteneurs du serveur de news, ils surveillent aussi ce que postes et que tu respectes les règles de base comme les crossposts (marteau-thérapie) \mbox{;-)}}

  \item[web@frankiz] {(\mail{web@frankiz}) Webmestres de frankiz, ils valident les annonces et les activités et surveillent le contenu du site de ton binet ou de ton site perso.}

  \item[X-share] {(\mail{xshare@frankiz}) Personne sympathique qui cherche à longueur de temps de nouveaux logiciels gratuits, ou mieux, libres, à proposer aux élèves dans \xshare.}

  \item[FAQ] {(\mail{faq@frankiz}) Ils sont là rassembler toutes les questions récurrentes pour remplir la FAQ (Foire aux Questions) de \fkz. N'hésite pas à leur donner des idées !}

  \item[InfoBR]{Boulet qui s'est trouvé un poste pipo et qui dit à tous les autres d'écrire. Si vous trouvez des bêtises ici, c'est de sa faute, donc vous pouvez aller le taper ;-)}

  \item[TV]{Changeurs de chaîne, dieux du multicast, ils ont souvent envie de taper les switches.}

  \item[QDJ Master (\mail{qdj@frankiz})]{Chaque jour un nouveau dilemne sur \fkz\ldots n'hésitez pas à faire vos propositions à \url{qdj@frankiz}.}

  \item[IRCop]{Responsable des relations avec RezoSup. Viendez sur IRC !}

  \item[TOL] {(\mail{tol@frankiz})Vérificateur de photos, il surveille le Trombi On Line.}

\end{description}
